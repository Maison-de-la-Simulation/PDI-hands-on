72  uint8_t* copy(uint8_t* to, const uint8_t* from)
73  {
74  to = reinterpret_cast<uint8_t*>(reinterpret_cast<uintptr_t>(to + m_align_mask) & ~m_align_mask);
75  if (m_subcontent.empty() && 1 == m_size) { // optimize for dense blocks
76  memcpy(to, from, m_repeat);
77  return static_cast<uint8_t*>(to) + m_repeat;
78  }
79  for (size_t ii = 0; ii < m_repeat; ++ii) {
80  if (m_subcontent.empty()) { // copy one byte
81  *to++ = *from;
82  }
83  for (auto&& mem : m_subcontent) {
84  to = mem.second.copy(to, from + mem.first);
85  }
86  from += m_size;
87  }
88  return to;
89  }
90 };

92 Data_layout desc(const Datatype& type)
93 {
94  if (auto&& scalar_type = dynamic_cast<const Scalar_datatype*>(&type)) {
95  return {scalar_type->datasize(), {}, 1, static_cast<uintptr_t>((1 << scalar_type->alignment()) - 1)};
96  }
97  if (auto&& array_type = dynamic_cast<const Array_datatype*>(&type)) {
98  Data_layout inner = desc(array_type->subtype());
99  size_t subsize = array_type->subsize();
100  size_t size = array_type->size();
101  if (subsize == size) { // dense array
102  inner.m_repeat *= subsize;
103  return inner;
104  } else { // sparse array
105  size_t buffer_size = inner.m_size * inner.m_repeat * size; // the contained buffer size is one element times the buffer size
106  inner.m_repeat *= subsize; // repeat the content as required
107  return {1, {make_pair(inner.m_size* inner.m_repeat * array_type->start(), move(inner))}, buffer_size, 0};
108  }
109  }
110  if (auto&& record_type = dynamic_cast<const Record_datatype*>(&type)) {
111  if (record_type->members().size() == 1) { // spurious optim when we have just a potentially sparsified version of the content
112  Data_layout inner = desc(record_type->members().front().type());
113  size_t offset = record_type->members().front().displacement();
114  size_t buffersize = record_type->buffersize();
115  if (inner.m_repeat == 1) { // we can sparsify by adding the offset to all subcontent
116  Data_layout result = {1, {}, record_type->buffersize(), 0};
117  for (auto&& sub : inner.m_subcontent) {
118  result.m_subcontent.emplace(sub.first + offset, move(sub.second)); // add the offset to all members
119  }
120  return result;
121  }
122  return {1, {make_pair(offset, move(inner))}, buffersize, 0};
123  }
124  Data_layout result = {1, {}, record_type->buffersize(), 0 };
125  for (auto&& mem : record_type->members()) {
126  Data_layout mem_layout = desc(mem.type());
127  if (mem_layout.m_repeat > 1) {
128  result.m_subcontent.emplace(mem.displacement(), move(mem_layout));
129  } else {
130  for (auto&& mem_subctnt : mem_layout.m_subcontent) {
131  mem_layout.m_subcontent.emplace(make_pair(mem.displacement() + mem_subctnt.first, move(mem_subctnt.second)));
132  }
133  }
134  }
135  return result;
136  }
137  assert(false && "Unsupported data type");
138 }

142 Ref Ref_base::do_copy(Ref_r ref)
143 {
144  Data_layout ref_desc = desc(ref.type());
145  void* newbuffer{operator new (ref_desc.m_size * ref_desc.m_repeat)};
146  try {

148  return Ref {newbuffer, [](void* d){::operator delete (d);}, ref.type().densify(), true, true};
149  } catch (...) {
