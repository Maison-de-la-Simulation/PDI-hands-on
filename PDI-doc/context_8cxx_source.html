57 void load_data(Context& ctx, PC_tree_t node, bool is_metadata)

68 plugin_loader_f PDI_NO_EXPORT get_plugin_ctr(const char* plugin_name)
69 {
70  string plugin_symbol = string{"PDI_plugin_"} + plugin_name + string{"_loader"};
71  void* plugin_ctor_uncast = dlsym(NULL, plugin_symbol.c_str());
72 
73  // case where the library was not prelinked
74  if (!plugin_ctor_uncast) {
75  string libname = string{"libpdi_"} + plugin_name + string{"_plugin.so"};
76  void* lib_handle = dlopen(libname.c_str(), RTLD_NOW);
77  if (!lib_handle) {
78  throw Error{PDI_ERR_PLUGIN, "Unable to load `%s' plugin file: %s", plugin_name, dlerror()};
79  }
80  plugin_ctor_uncast = dlsym(lib_handle, plugin_symbol.c_str());
81  }
82 
83  if (!plugin_ctor_uncast) {
84  throw Error{PDI_ERR_PLUGIN, "Unable to load `%s' plugin from file: %s", plugin_name, dlerror()};
85  }
86 
87  return reinterpret_cast<plugin_loader_f>(plugin_ctor_uncast);
88 }

121 Context::Context(PC_tree_t conf, MPI_Comm* world)
122 {
123  // no metadata is not an error
124  PC_tree_t metadata = PC_get(conf, ".metadata");
125  if (!PC_status(metadata)) {
126  load_data(*this, metadata, true);
127  }
128 
129  // no data is spurious, but not an error
130  PC_tree_t data = PC_get(conf, ".data");
131  if (!PC_status(data)) {
132  load_data(*this, data, false);
133  }
134 
135  int nb_plugins = len(PC_get(conf, ".plugins"), 0);
136  for (int plugin_id = 0; plugin_id < nb_plugins; ++plugin_id) {
137  //TODO: what to do if a single plugin fails to load?
138  string plugin_name = to_string(PC_get(conf, ".plugins{%d}", plugin_id));
139  try {
140  plugins.emplace(plugin_name, get_plugin_ctr(plugin_name.c_str())(*this, PC_get(conf, ".plugins<%d>", plugin_id), world));
141  } catch (const exception& e) {
142  throw Error{PDI_ERR_SYSTEM, "Error while loading plugin `%s': %s", plugin_name.c_str(), e.what()};
143  }
144  }
145 }

177 void Context::event(const char* name)
178 {
179  for (auto& elmnt : plugins) {
180  vector<Error> errors;
181  try {
182  elmnt.second->event(name);
183  //TODO: remove the faulty plugin in case of error?
184  } catch (const Error& e) {
185  errors.emplace_back(e.status(), "for plugin `%s': %s", elmnt.first.c_str(), e.what());
186  } catch (const std::exception& e) {
187  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s': %s", elmnt.first.c_str(), e.what());
188  } catch (...) {
189  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s'", elmnt.first.c_str());
190  }
191  if ( 1 == errors.size() ) throw Error{errors.front().status(), "While triggering named event `%s' %s", name, errors.front().what()};
192  if ( !errors.empty() ) {
193  string errmsg = "multiple errors while triggering named event `";
194  errmsg += name;
195  errmsg += "':";
196  for ( auto&& err: errors ) {
197  errmsg += " ";
198  errmsg += err.what();
199  }
200  throw Error{PDI_ERR_PLUGIN, "%s", errmsg.c_str()};
201  }
202  }
203 }
