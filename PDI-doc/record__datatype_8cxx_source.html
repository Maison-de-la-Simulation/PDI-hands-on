93 Datatype_uptr Record_datatype::densify() const
94 {
95  long displacement = 0;
96  vector<Record_datatype::Member> densified_members;
97  for (auto&& member : m_members) {
98  densified_members.emplace_back(displacement, member.type().densify(), member.name());
99  displacement += densified_members.back().type().datasize();
100  }
101  return unique_ptr<Record_datatype> {new Record_datatype{move(densified_members), m_buffersize}};
102 }
