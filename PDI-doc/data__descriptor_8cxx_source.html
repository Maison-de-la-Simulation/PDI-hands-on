91 Data_descriptor::~Data_descriptor()
92 {
93  // release metadata copies we kept
94  if ( metadata() && m_refs.size() == 1) m_refs.pop();
95 
96  /* on error, we might be destroyed while not empty. In that case, don't keep
97  * ownership
98  */
99  if (!m_refs.empty()) {
100  std::cerr<<" *** [PDI] Warning: Remaining "<<m_refs.size()<<" reference(s) to `"<<m_name<<"' in PDI after program end"<<std::endl;
101  }
102  if ( metadata() ) while (!m_refs.empty()) m_refs.pop();
103  if ( !metadata() ) while (!m_refs.empty()) reclaim();
104 }

143 void* Data_descriptor::share(Ref data_ref, bool read, bool write)
144 {
145  // metadata must provide read access
146  if (metadata() && !Ref_r(data_ref)) {
147  throw Error{PDI_ERR_RIGHT, "Metadata sharing must offer read access"};
148  }
149 
150  // for metadata, unlink happens on share
151  if (!m_refs.empty() && metadata()) {
152  m_refs.pop();
153  }
154 
155  // make a reference and put it in the store
156  void* result = nullptr;
157  if (read) {
158  if (write) {
159  result = Ref_rw{data_ref} .get(nothrow);
160  m_refs.emplace(new Ref_holder::Impl<true, true>(data_ref));
161  } else {
162  result = const_cast<void*>(Ref_r{data_ref} .get(nothrow));
163  m_refs.emplace(new Ref_holder::Impl<true, false>(data_ref));
164  }
165  } else {
166  if (write) {
167  result = Ref_w{data_ref} .get(nothrow);
168  m_refs.emplace(new Ref_holder::Impl<false, true>(data_ref));
169  } else {
170  m_refs.emplace(new Ref_holder::Impl<false, false>(data_ref));
171  }
172  }
173 
174  if (data_ref && !ref()) {
175  m_refs.pop();
176  throw Error{PDI_ERR_RIGHT, "Unable to grant requested rights"};
177  }
178 
179  for (auto& elmnt : m_context.plugins) {
180  vector<Error> errors;
181  try {
182  elmnt.second->data(m_name.c_str(), ref());
183  //TODO: remove the faulty plugin in case of error?
184  } catch (const Error& e) {
185  errors.emplace_back(e.status(), "for plugin `%s': %s", elmnt.first.c_str(), e.what());
186  } catch (const std::exception& e) {
187  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s': %s", elmnt.first.c_str(), e.what());
188  } catch (...) {
189  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s'", elmnt.first.c_str());
190  }
191  if ( 1 == errors.size() ) throw Error{errors.front().status(), "While triggering data event `%s' %s", m_name.c_str(), errors.front().what()};
192  if ( !errors.empty() ) {
193  string errmsg = "multiple errors while triggering data event `"+m_name+"':";
194  for ( auto&& err: errors ) {
195  errmsg += " ";
196  errmsg += err.what();
197  }
198  throw Error{PDI_ERR_PLUGIN, "%s", errmsg.c_str()};
199  }
200  }
201 
202  return result;
203 }

205 void Data_descriptor::release()
206 {
207  // move reference out of the store
208  if (m_refs.empty()) throw Error{PDI_ERR_VALUE, "Cannot release a non shared value"};
209 
210  Ref oldref = ref();
211  m_refs.pop();
212  if (metadata()) {
213  // re-share ourselves & keep a read ref to be used
214  m_refs.emplace(new Ref_holder::Impl<true, false>(oldref));
215  }
216 }

218 void Data_descriptor::reclaim()
219 {
220  if (m_refs.empty()) throw Error{PDI_ERR_VALUE, "Cannot reclaim a non shared value"};
221 
222  Ref oldref = ref();
223  m_refs.pop();
224  if (metadata()) {
225  // if the content is a metadata, keep a copy
226  m_refs.emplace(new Ref_holder::Impl<true, false>(oldref.copy()));
227  }
228  oldref.release();
229 }
