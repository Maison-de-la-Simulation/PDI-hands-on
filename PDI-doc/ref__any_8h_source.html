420  void PDI_NO_EXPORT link(Referenced* content) noexcept
421  {
422  assert(!m_content);
423  if (!content || !content->m_buffer) return; // null ref
424  if (R && content->m_read_locks) return;
425  if (W && content->m_write_locks) return;
