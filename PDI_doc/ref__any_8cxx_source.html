44 Ref Ref_base::do_copy(Ref_r ref)
45 {
46  Datatype_uptr densified_type {ref.type().densify()};
47  if ( !densified_type->buffersize() ) {
48  return Ref {};
49  }
50  // no std::aligned_alloc or std::align_val_t in C++14, hand-written version
51  // size + (densified_type->alignment() - 1) <- we want to make sure that we fit the data even though the worst alignment occur
52  size_t size = densified_type->buffersize() + (densified_type->alignment() - 1);
53  void* buffer = operator new (size);
54  void* data = std::align(densified_type->alignment(), densified_type->buffersize(), buffer, size);
55  try {
56  ref.type().data_to_dense_copy(data, ref.get());
57  } catch (...) {
58  ::operator delete (buffer);
59  throw;
60  }
61  return Ref {data, [buffer](void*){operator delete (buffer);}, std::move(densified_type), true, true};
62 }
