151 void* Scalar_datatype::data_to_dense_copy(void* to, const void* from) const
152 {
153  if ( !m_copy ) {
154  memcpy(to, from, datasize());
155  to = reinterpret_cast<uint8_t*>(to) + datasize();
156  } else {
