132 void* Array_datatype::data_to_dense_copy(void* to, const void* from) const
133 {
134  size_t subtype_buffersize = subtype().buffersize();
135  from = static_cast<const uint8_t*>(from) + (start() * subtype_buffersize);
136 
137  if (subtype().simple() && subtype().dense()) {
138  //dense copy
139  memcpy(to, from, datasize());
140  return reinterpret_cast<uint8_t*>(to) + datasize();
141  }
142 
143  size_t subtype_alignment = subtype().alignment();
144  for (size_t subtype_no = 0; subtype_no < subsize(); subtype_no++) {
145  //space_to_align is set to alignment(), because we always find the alignment in the size of alignment
146  size_t space_to_align = subtype_alignment;
147  //size = 0, because we know that to points to allocated memory
148  to = std::align(subtype_alignment, 0, to, space_to_align);
149 
150  to = subtype().data_to_dense_copy(to, from);
151  from = reinterpret_cast<const uint8_t*>(from) + subtype_buffersize;
152  }
153  return to;
154 }

156 void* Array_datatype::data_from_dense_copy(void* to, const void* from) const
157 {
158  uint8_t* original_to = reinterpret_cast<uint8_t*>(to);
159  size_t subtype_buffersize = subtype().buffersize();
160  to = static_cast<uint8_t*>(to) + (start() * subtype_buffersize);
161 
162  if (subtype().simple() && subtype().dense()) {
163  //dense copy
164  memcpy(to, from, datasize());
165  to = original_to + buffersize();
166  return to;
167  }
168 
169  size_t subtype_alignment = subtype().alignment();
170  for (int subtype_no = 0; subtype_no < subsize(); subtype_no++) {
171  //space_to_align is set to alignment(), because we always find the alignment in the size of alignment
172  size_t space_to_align = subtype_alignment;
173  //size = 0, because we know that to points to allocated memory
174  to = std::align(subtype_alignment, 0, to, space_to_align);
175 
176  //cannot use std::aling, becasue `from' is const
177  auto subtype_align = subtype().alignment();
178  int padding = (subtype_align - (reinterpret_cast<const uintptr_t>(from) % subtype_align)) % subtype_align;
179  from = reinterpret_cast<const uint8_t*>(from) + padding;
180 
181  to = subtype().data_from_dense_copy(to, from);
182  from = reinterpret_cast<const uint8_t*>(from) + subtype().datasize();
183  }
184  to = original_to + buffersize();
185  return to;
186 }

188 void Array_datatype::destroy_data(void* ptr) const
189 {

192  for (size_t subtype_no = start(); subtype_no < start()+subsize(); ++subtype_no) {
