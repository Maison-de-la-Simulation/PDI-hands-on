62 Logger_sptr select_log_sinks(PC_tree_t logging_tree)
63 {
64  vector<sink_ptr> sinks;
65  PC_tree_t output_tree = PC_get(logging_tree, ".output");
66 
67  //configure file sink
68  if (!PC_status(PC_get(output_tree, ".file"))) {
69  string filename {to_string(PC_get(output_tree, ".file"))};
70  auto file_sink = make_shared<basic_file_sink_st>(filename);
71  sinks.emplace_back(file_sink);
72  }
73 
74  //configure console sink
75  if (
76  (!PC_status(PC_get(output_tree, ".console")) || sinks.empty() ) // either there is a console sink specified or no other
77  && to_string(PC_get(output_tree, ".console"), "on") != "off" // the console sink is not specifically disabled
78  ) {
79  //logging to console is turned on
80 #if defined _WIN32 && !defined(__cplusplus_winrt)
81  sinks.push_back(make_shared<wincolor_stdout_sink_st>());
82 #else
83  sinks.push_back(make_shared<ansicolor_stdout_sink_st>());
84 #endif
85  }
86 
87  return make_shared<logger>("PDI_logger", sinks.begin(), sinks.end());
88 }

94 void read_log_level(Logger_sptr logger, PC_tree_t logging_tree)
95 {
96  if (!PC_status(PC_get(logging_tree, ".level"))) {
97  string level_str {to_string(PC_get(logging_tree, ".level"))};
98 
99  unordered_map<string, level_enum> level_map = {
100  {"debug", debug},
101  {"info", info},
102  {"warn", warn},
103  {"error", err},
104  {"off", off}
105  };
106 
107  auto level_it = level_map.find(level_str);
108  if (level_it != level_map.end()) {
109  logger->set_level(level_map.find(level_str)->second);
110  } else {
111  logger->warn("Invalid logging level: {}. Available: 'debug', 'info', 'warn', 'error', 'off'.", level_str);
112  }
113  }
114 }
