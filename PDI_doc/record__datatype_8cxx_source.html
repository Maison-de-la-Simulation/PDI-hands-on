112 Datatype_uptr Record_datatype::densify() const
113 {
114  size_t displacement = 0;
115  vector<Record_datatype::Member> densified_members;
116  for (auto&& member : m_members) {
117  Datatype_uptr densified_type = member.type().densify();
118  size_t alignment = densified_type->alignment();
119  // align the next member as requested
120  displacement += (alignment - (displacement % alignment)) % alignment;
121  densified_members.emplace_back(displacement, move(densified_type), member.name());
122  displacement += densified_members.back().type().buffersize();
123  }
124  //add padding at the end of record
125  size_t record_alignment = alignment();
126  displacement += (record_alignment - (displacement % record_alignment)) % record_alignment;
127 
128  // ensure the record size is at least 1 to have a unique address
129  displacement = max<size_t>(1, displacement);
130  return unique_ptr<Record_datatype> {new Record_datatype{move(densified_members), displacement}};
131 }

138 bool Record_datatype::dense() const
139 {
140  size_t displacement = 0;
141  for (auto&& member : m_members) {
142  if ( !member.type().dense() ) return false;
143  size_t alignment = member.type().alignment();
144  // add space for alignment
145  displacement += (alignment - (displacement % alignment)) % alignment;
146  if ( member.displacement() > displacement ) return false;
147  displacement += member.type().buffersize();
148  }
149  //add padding at the end of record
150  size_t record_alignment = alignment();
151  displacement += (record_alignment - (displacement % record_alignment)) % record_alignment;
152 
153  // accept 1 extra byte for unique address
154  displacement = max<size_t>(1, displacement);
155  if ( buffersize() > displacement ) return false;
156  return true;
157 }

190 void* Record_datatype::data_to_dense_copy(void* to, const void* from) const
191 {
192  if (simple() && dense()) {
193  //dense copy
194  memcpy(to, from, buffersize());
195  to = reinterpret_cast<uint8_t*>(to) + buffersize();
196  return to;
197  }
198 
199  for (auto&& member : members()) {
200  //space_to_align is set to alignment(), because we always find the alignment in the size of alignment
201  auto space_to_align = member.type().alignment();
202  //size = 0, because we know that to points to allocated memory
203  to = std::align(member.type().alignment(), 0, to, space_to_align);
204  const uint8_t* member_from = reinterpret_cast<const uint8_t*>(from) + member.displacement();
205  to = member.type().data_to_dense_copy(to, member_from);
206  }
207  return to;
208 }

210 void* Record_datatype::data_from_dense_copy(void* to, const void* from) const
211 {
212  uint8_t* original_to = reinterpret_cast<uint8_t*>(to);
213 
214  if (simple() && dense()) {
215  //dense copy
216  memcpy(to, from, datasize());
217  to = original_to + buffersize();
218  return to;
219  }
220 
221  for (auto&& member : members()) {
222  auto member_align = member.type().alignment();
223  int padding = (member_align - (reinterpret_cast<const uintptr_t>(from) % member_align)) % member_align;
224  from = reinterpret_cast<const uint8_t*>(from) + padding;
225  to = original_to + member.displacement();
226 
227  member.type().data_from_dense_copy(to, from);
228  from = reinterpret_cast<const uint8_t*>(from) + member.type().datasize();
229  }
230  to = original_to + buffersize();
231  return to;
232 }
