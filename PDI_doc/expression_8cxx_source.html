271 unique_ptr<Expression::Impl> Expression::Impl::parse_term(char const** val_str)
272 {
273  if (**val_str == '(') {
274  const char* term = *val_str;
275  ++term;
276  while (isspace(*term)) ++term;
277  unique_ptr<Expression::Impl> result = Operation::parse(&term, 1);
278  if (*term != ')') throw Error {PDI_ERR_VALUE, "Expected ')', found '%c'", *term};
279  ++term;
280  while (isspace(*term)) ++term;
281  *val_str = term;
282  return result;
283  } else if (**val_str == '$') {
284  return Reference_expression::parse(val_str);
285  } else {
286  return Int_literal::parse(val_str);
287  }

291 {
292  const char* id = *val_str;
293 
294  if (!(

296  || (*id >= 'A' && *id <= 'Z')
297  || (*id == '_')
298  )) {
299  throw Error {PDI_ERR_VALUE, "Invalid first ID character: %c", *id};

303 
304  while (
305  (*id >= 'a' && *id <= 'z')
306  || (*id >= 'A' && *id <= 'Z')

309  ) {
310  ++(id_len);
311  ++id;
312  }
313 
314  string result { *val_str, id_len };
315  *val_str = id;

398 unique_ptr<Expression::Impl> Expression::Impl::String_literal::parse(char const** val_str)
399 {
400  const char* str = *val_str;
401 
402  unique_ptr<String_literal> result{new String_literal};
403 
404  string* curstr = &result->start;
405  while (*str) {
406  size_t sz = 0;
407  while (str[sz] != '\\' && str[sz] != '$' && str[sz]) ++sz;
408  curstr->append(str, sz);
409  str += sz;
410  switch (*str) {
411  case '\\': {
412  str += 2;
413  curstr->push_back('\\');
414  } break;
415  case '$': {
416  switch (str[1]) {
417  case '(': { // remove the dollar, parse the term starting with the parenthesis (the operation)
418  ++str;
419  result->values.emplace_back(Expression{parse_term(&str)}, "");
420  curstr = &result->values.back().second;
421  } break;
422  default: { // parse the term starting with the dollar (the ref)
423  result->values.emplace_back(Expression{Reference_expression::parse(&str)}, "");
424  curstr = &result->values.back().second;
425  } break;
426  }
427  } break;
428  case 0: {} break;
429  default: {
430  throw Error {PDI_ERR_IMPL, "Unexpected error!!!"};
431  }
432  }
433  }
434 
435  *val_str = str;
436  return result;
437 }

444 long Expression::Impl::Reference_expression::to_long(Context& ctx) const
445 try
446 {
447  if (Ref_r ref = ctx.desc(m_referenced.c_str()).ref()) {
448  const Datatype* type = &ref.type();
449  long stride = 1;
450  long idx = 0;
451  for (auto&& ii : m_idx) {
452  auto&& array_type = dynamic_cast<const Array_datatype*>(type);
453  if (!array_type) throw Error {PDI_ERR_VALUE, "Accessing non-array data with an index"};
454  idx += (array_type->start() + ii.to_long(ctx)) * stride;
455  stride *= array_type->size();
456  type = &array_type->subtype();
457  }
458 
459  auto&& scalar_type = dynamic_cast<const Scalar_datatype*>(type);
460  if (!scalar_type) throw Error {PDI_ERR_VALUE, "Expected scalar found invalid type instead"};
461 
462  if (scalar_type->kind() == Scalar_kind::SIGNED) {
463  switch (scalar_type->datasize()) {
464  case 1:
465  return static_cast<const int8_t*>(ref.get())[idx];
466  case 2:
467  return static_cast<const int16_t*>(ref.get())[idx];
468  case 4:
469  return static_cast<const int32_t*>(ref.get())[idx];
470  case 8:
471  return static_cast<const int64_t*>(ref.get())[idx];
472  default:
473  throw Error(PDI_ERR_VALUE, "Unexpected int size: %ld", static_cast<long>(scalar_type->kind()));
474  }
475  } else if (scalar_type->kind() == Scalar_kind::UNSIGNED) {
476  switch (scalar_type->datasize()) {
477  case 1:
478  return static_cast<const uint8_t*>(ref.get())[idx];
479  case 2:
480  return static_cast<const uint16_t*>(ref.get())[idx];
481  case 4:
482  return static_cast<const uint32_t*>(ref.get())[idx];
483  case 8:
484  return static_cast<const uint64_t*>(ref.get())[idx];
485  default:
486  throw Error(PDI_ERR_VALUE, "Unexpected uint size: %ld", static_cast<long>(scalar_type->kind()));
487  }
488  }
489  throw Error {PDI_ERR_VALUE, "Expected integer scalar"};
490  }
491  throw Error {PDI_ERR_RIGHT, "Unable to grant access for value reference"};
492 } catch (const Error& e)
493 {
494  throw Error {e.status(), "while referencing `%s': %s", m_referenced.c_str(), e.what()};
495 }

511 unique_ptr<Expression::Impl> Expression::Impl::Reference_expression::parse(char const** val_str)
512 {
513  const char* ref = *val_str;
514  unique_ptr<Reference_expression> result{new Reference_expression};
515 
516  if (*ref != '$') throw Error {PDI_ERR_VALUE, "Expected '$', got %c", *ref};
517  ++ref;
518 
519  bool has_curly_brace = false;
520  if (*ref == '{') {
521  ++ref;
522  has_curly_brace = true;
523  while (isspace(*ref)) ++ref;
524  }
525 
526  result->m_referenced = parse_id(&ref);
527 
528  while (isspace(*ref)) ++ref;
529 
530  while (*ref == '[') {
531  ++ref;
532  while (isspace(*ref)) ++ref;
533  result->m_idx.emplace_back(Expression{Operation::parse(&ref, 1)});
534  if (*ref != ']') {
535  throw Error {PDI_ERR_VALUE, "Expected ']', found %c", *ref};
536  }
537  ++ref;
538  while (isspace(*ref)) ++ref;
539  }
540 
541  if (has_curly_brace) {
542  if (*ref != '}') {
543  throw Error {PDI_ERR_VALUE, "Expected '}', found %c", *ref};
544  }
545  ++ref;
546  while (isspace(*ref)) ++ref;
547  }
548 
549  *val_str = ref;
550  return result;
551 }

553 long Expression::Impl::Operation::to_long(Context& ctx) const
554 {
555  long computed_value = m_first_operand.to_long(ctx);
556  for (auto&& op: m_operands) {
557  long operand = op.second.to_long(ctx);
558  switch (op.first) {
559  case PLUS: {
560  computed_value += operand;
561  } break;
562  case MINUS: {
563  computed_value -= operand;
564  } break;
565  case MULT: {
566  computed_value *= operand;
567  } break;
568  case DIV: {
569  computed_value /= operand;
570  } break;
571  case MOD: {
572  computed_value %= operand;
573  } break;
574  case EQUAL: {
575  computed_value = (computed_value == operand);
576  } break;
577  case AND: {
578  computed_value = computed_value && operand;
579  } break;
580  case OR: {
581  computed_value = computed_value || operand;
582  } break;
583  case GT: {
584  computed_value = (computed_value > operand);
585  } break;
586  case LT: {
587  computed_value = (computed_value < operand);
588  } break;
589  }
590  }
591 
592  return computed_value;
593 }

611 unique_ptr<Expression::Impl> Expression::Impl::Operation::parse(char const** val_str, int level)
612 {
613  // a level 7 operation is a term
614  if (level == 7) return parse_term(val_str);
615 
616  const char* exprval = *val_str;
617  unique_ptr<Expression::Impl> result = parse(&exprval, level + 1);
618 
619  // we only build the Operation if needed, otherwise we return the previous
620  // expression directly
621  unique_ptr<Operation> expr = NULL;
622  while (op_level(exprval) == level) {
623  if (!expr) {
624  expr.reset(new Operation);
625  expr->m_first_operand = move(result);
626  }
627  Operator oper = parse_operator(&exprval, level);
628  unique_ptr<Expression::Impl> operand = parse(&exprval, level + 1);
629  expr->m_operands.emplace_back(oper, Expression{move(operand)});
630  }
631 
632  *val_str = exprval;
633  if (expr) {
634  return expr;
635  } else {
636  return result;
637  }
638 }

653 Expression::Impl::Operation::Operator Expression::Impl::Operation::parse_operator(char const** val_str, int level)
654 {
655  const char* c_op = *val_str;
656  int found_level = op_level(c_op);
657  if (found_level == 0) {
658  throw Error {PDI_ERR_VALUE, "Expected operator, found '%c'", *c_op};
659  }
660  if (found_level != level) {
661  throw Error {PDI_ERR_VALUE, "Mixing operator priority"};
662  }
663  Operator op = static_cast<Operator>(*c_op);
664  ++c_op;
665 
666  while (isspace(*c_op)) ++c_op;
667 
668  *val_str = c_op;
669  return op;
670 }
