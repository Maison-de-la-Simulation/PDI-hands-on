69 pytup pyversion(pytup pyexpected)
70 {
71  Paraconf_wrapper fw;
72 
73  unsigned long expected_major = pyexpected[0].cast<unsigned long>();
74  unsigned long expected_minor = pyexpected[1].cast<unsigned long>();
75  unsigned long expected_patch = pyexpected[2].cast<unsigned long>();
76 
77  if (
78  ( expected_major || expected_minor || expected_patch )
79  && (
80  expected_major != PDI_VERSION_MAJOR
81  || expected_minor > PDI_VERSION_MINOR
82  )
83  ) {
84  throw Error{
85  PDI_ERR_PLUGIN,
86  "Invalid PDI API version: %lu.%lu.%lu, PDI provided version is %lu.%lu.%lu",
87  expected_major,
88  expected_minor,
89  expected_patch,
90  PDI_VERSION_MAJOR,
91  PDI_VERSION_MINOR,
92  PDI_VERSION_PATCH
93  };
94  }
95 
96  return pybind11::make_tuple(PDI_VERSION_MAJOR, PDI_VERSION_MINOR, PDI_VERSION_PATCH);
97 }
