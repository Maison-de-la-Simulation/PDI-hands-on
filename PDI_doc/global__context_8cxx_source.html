59 void load_data(Context& ctx, PC_tree_t node, bool is_metadata)

63  for (int map_id = 0; map_id < map_len; ++map_id) {
64  Data_descriptor& dsc = ctx.desc(to_string(PC_get(node, "{%d}", map_id)).c_str());
65  dsc.metadata(is_metadata);
66  dsc.default_type(ctx.datatype(PC_get(node, "<%d>", map_id)));
67  }

70 plugin_loader_f PDI_NO_EXPORT get_plugin_ctr(const char* plugin_name)
71 {
72  string plugin_symbol = string{"PDI_plugin_"} + plugin_name + string{"_loader"};
73  void* plugin_ctor_uncast = dlsym(NULL, plugin_symbol.c_str());
74 
75  // case where the library was not prelinked
76  if (!plugin_ctor_uncast) {
77  string libname = string{"libpdi_"} + plugin_name + string{"_plugin.so"};
78  void* lib_handle = dlopen(libname.c_str(), RTLD_NOW);
79  if (!lib_handle) {
80  throw Error{PDI_ERR_PLUGIN, "Unable to load `%s' plugin file: %s", plugin_name, dlerror()};
81  }
82  plugin_ctor_uncast = dlsym(lib_handle, plugin_symbol.c_str());
83  }
84 
85  if (!plugin_ctor_uncast) {
86  throw Error{PDI_ERR_PLUGIN, "Unable to load `%s' plugin from file: %s", plugin_name, dlerror()};
87  }
88 
89  return reinterpret_cast<plugin_loader_f>(plugin_ctor_uncast);
90 }

117 Global_context::Global_context(PC_tree_t conf):
118  m_logger{configure_logger(conf)}
119 {
120  // load basic datatypes
121  Datatype_template::load_basic_datatypes(*this);
122 
123  // first load plugins, because they can add datatypes declared in config
124  int nb_plugins = len(PC_get(conf, ".plugins"), 0);
125  for (int plugin_id = 0; plugin_id < nb_plugins; ++plugin_id) {
126  //TODO: what to do if a single plugin fails to load?
127  string plugin_name = to_string(PC_get(conf, ".plugins{%d}", plugin_id));
128  try {
129  m_plugins.emplace(plugin_name, get_plugin_ctr(plugin_name.c_str())(*this, PC_get(conf, ".plugins<%d>", plugin_id), m_logger));
130  } catch (const exception& e) {
131  throw Error{PDI_ERR_SYSTEM, "Error while loading plugin `%s': %s", plugin_name.c_str(), e.what()};
132  }
133  }
134 
135  // no metadata is not an error
136  PC_tree_t metadata = PC_get(conf, ".metadata");
137  if (!PC_status(metadata)) {
138  load_data(*this, metadata, true);
139  }
140 
141  // no data is spurious, but not an error
142  PC_tree_t data = PC_get(conf, ".data");
143  if (!PC_status(data)) {
144  load_data(*this, data, false);
145  }
146 }

178 void Global_context::event(const char* name)
179 {
180  for (auto& elmnt : m_plugins) {
181  vector<Error> errors;
182  try {
183  elmnt.second->event(name);
184  //TODO: remove the faulty plugin in case of error?
185  } catch (const Error& e) {
186  errors.emplace_back(e.status(), "for plugin `%s': %s", elmnt.first.c_str(), e.what());
187  } catch (const exception& e) {
188  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s': %s", elmnt.first.c_str(), e.what());
189  } catch (...) {
190  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s'", elmnt.first.c_str());
191  }
192  if ( 1 == errors.size() ) throw Error{errors.front().status(), "While triggering named event `%s' %s", name, errors.front().what()};
193  if ( !errors.empty() ) {
194  string errmsg = "multiple errors while triggering named event `";
195  errmsg += name;
196  errmsg += "':";
197  for ( auto&& err: errors ) {
198  errmsg += " ";
199  errmsg += err.what();
200  }
201  throw Error{PDI_ERR_PLUGIN, "%s", errmsg.c_str()};
202  }
203  }

211 Datatype_template_uptr Global_context::datatype(PC_tree_t node)
212 {
213  char* type_c;
214  if ( PC_string(PC_get(node, ".type"), &type_c) ) {
215  if ( PC_string(node, &type_c) ) {
216  throw Error{PDI_ERR_TYPE, "Invalid type descriptor"};
217  }
218  }
219  string type = type_c;
220 
221  // check if someone didn't mean to create an array with the old syntax
222  if ( type != "array" && !PC_status(PC_get(node, ".size"))) {
223  logger()->warn("Non-array type with a `size' property");
224  }
225 
226  auto func_it = m_datatype_parsers.find(type);
227  if (func_it != m_datatype_parsers.end()) {
228  return (func_it->second)(*this, node);
229  }
230  throw Error{PDI_ERR_TYPE, "Cannot find datatype `%s'", type.c_str()};
231 }
