178 vector<Expression> get_array_property(PC_tree_t node, string property)
179 {
180  vector<Expression> prop_vector;
181  PC_tree_t config = PC_get(node, property.c_str());
182  if (!PC_status(PC_get(config, "[0]")) ) {
183  int rank = len(config);
184  for (int i = 0; i < rank; i++) {
185  prop_vector.emplace_back(to_string(PC_get(node, (property + string("[%d]")).c_str(), i)));
186  }
187  } else {
188  string prop = to_string(config, "");
189  if (prop.length() > 0) {
190  prop_vector.emplace_back(prop);
191  }
192  }
193  return prop_vector;
194 }

196 void validate_array(vector<Expression>& size, vector<Expression>& subsize, vector<Expression>& start)
197 {
198  if (size.empty()) {
199  throw Error{PDI_ERR_CONFIG, "Array must have defined `size'"};
200  }
201  if (!start.empty() && subsize.empty()) {
202  throw Error{PDI_ERR_CONFIG, "Array with a `start` property must have a defined `subsize'"};
203  //TODO: handle by setting subsize to size-start
204  }
205  if (start.empty()) {
206  start = vector<Expression>(size.size(), Expression{0l});
207  }
208  if (subsize.empty()) {
209  subsize = size;
210  }
211 
212  //check if rank of array is correct
213  if (size.size() != subsize.size()) {
214  throw Error{PDI_ERR_CONFIG, "`subsize' must have the same rank as `size': %d != %d", subsize.size(), size.size()};
215  }
216  if (size.size() != start.size()) {
217  throw Error{PDI_ERR_CONFIG, "`start' must have the same rank as `size': %d != %d", start.size(), size.size()};
218  }
219 }

223 Datatype_template_uptr to_array_datatype_template(Context& ctx, PC_tree_t node)
224 {
225  {
226  string order_str = to_string(PC_get(node, ".order"), "");
227  if (order_str == "c" && order_str == "C") {
228  ctx.logger()->warn("`order: C' for array is the only supported order and its specification is deprecated");
229  } else if (order_str !="" ) {
230  throw Error{PDI_ERR_CONFIG, "Incorrect array ordering: `%s', only C order is supported", order_str.c_str()};
231  }
232  }
233 
234  vector<Expression> array_size = get_array_property(node, ".size");
235  vector<Expression> array_subsize = get_array_property(node, ".subsize");
236  vector<Expression> array_start = get_array_property(node, ".start");
237 
238  validate_array(array_size, array_subsize, array_start);
239 
240  PC_tree_t config_elem = PC_get(node, ".subtype");
241  if (PC_status(config_elem)) {
242  throw Error{PDI_ERR_CONFIG, "Array must have `subtype'"};
243  }
244 
245  Datatype_template_uptr res_type = ctx.datatype(config_elem);
246 
247  for (ssize_t ii = array_size.size()-1; ii >=0; --ii) {
248  res_type.reset(new Array_template(move(res_type), move(array_size[ii]), move(array_start[ii]), move(array_subsize[ii])));
249  }
250  return res_type;
251 }

253 vector<Record_template::Member> get_members(Context& ctx, PC_tree_t member_list_node)
254 {
255  vector<Record_template::Member> members;
256 
257  int nb_members = len(member_list_node, 0);
258  for (int member_id = 0; member_id < nb_members; member_id++) {
259  //get current member name
260  string member_name = to_string(PC_get(member_list_node, "{%d}", member_id));
261 
262  //get current member
263  PC_tree_t member_node = PC_get(member_list_node, "<%d>", member_id);
264 
265  PC_tree_t disp_conf = PC_get(member_node, ".disp");
266  if (PC_status(disp_conf)) {
267  throw Error{PDI_ERR_CONFIG, "All members must have displacements"};
268  }
269  Expression disp = to_string(disp_conf);
270 
271  members.emplace_back(move(disp), ctx.datatype(member_node), move(member_name));
272  }
273  return members;
274 }
