91 Data_descriptor_impl::~Data_descriptor_impl()
92 {
93  // release metadata placeholder
94  if ( metadata() && m_refs.size() == 1) m_refs.pop();
95 
96  // on error, we might be destroyed while not empty.
97  if (!m_refs.empty()) {
98  m_context.logger()->warn("Remaining {} reference(s) to `{}' in PDI after program end", m_refs.size()-(metadata()?1:0), m_name);
99  // leak the remaining data
100  while ( !m_refs.empty() ) {
101  m_refs.top().release();
102  m_refs.pop();
103  }
104  }

144 Ref Data_descriptor_impl::ref()
145 {
146  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
147  if (m_refs.empty()) {
148  for (auto& elmnt : m_context.m_plugins) {
149  vector<Error> errors;
150  try {
151  elmnt.second->empty_desc_access(m_name.c_str());
152  //TODO: remove the faulty plugin in case of error?
153  } catch (const Error& e) {
154  errors.emplace_back(e.status(), "for plugin `%s': %s", elmnt.first.c_str(), e.what());
155  } catch (const exception& e) {
156  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s': %s", elmnt.first.c_str(), e.what());
157  } catch (...) {
158  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s'", elmnt.first.c_str());
159  }
160  if ( 1 == errors.size() ) throw Error{errors.front().status(), "While triggering empty desc access `%s' %s", m_name, errors.front().what()};
161  if ( !errors.empty() ) {
162  string errmsg = "multiple errors while triggering empty desc access`";
163  errmsg += m_name;
164  errmsg += "':";
165  for ( auto&& err: errors ) {
166  errmsg += " ";
167  errmsg += err.what();
168  }
169  throw Error{PDI_ERR_PLUGIN, "%s", errmsg.c_str()};
170  }
171  }
172 
173  //at least one plugin should share a Ref
174  if (m_refs.empty()) {
175  throw Error{PDI_ERR_VALUE, "Cannot access a non shared value: `%s'", m_name.c_str()};
176  }
177  }
178  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
179  return m_refs.top()->ref();
180 }

203 void* Data_descriptor_impl::share(Ref data_ref, bool read, bool write)
204 {
205  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
206  // metadata must provide read access
207  if (metadata() && !Ref_r(data_ref)) {
208  throw Error{PDI_ERR_RIGHT, "Metadata sharing must offer read access"};
209  }
210 
211  // make a reference and put it in the store
212  void* result = nullptr;
213  if (read) {
214  if (write) {
215  result = Ref_rw{data_ref} .get(nothrow);
216  m_refs.emplace(new Ref_holder::Impl<true, true>(data_ref));
217  } else {
218  result = const_cast<void*>(Ref_r{data_ref} .get(nothrow));
219  m_refs.emplace(new Ref_holder::Impl<true, false>(data_ref));
220  }
221  } else {
222  if (write) {
223  result = Ref_w{data_ref} .get(nothrow);
224  m_refs.emplace(new Ref_holder::Impl<false, true>(data_ref));
225  } else {
226  m_refs.emplace(new Ref_holder::Impl<false, false>(data_ref));
227  }
228  }
229 
230  if (data_ref && !ref()) {
231  m_refs.pop();
232  throw Error{PDI_ERR_RIGHT, "Unable to grant requested rights"};
233  }
234 
235  for (auto& elmnt : m_context.m_plugins) {
236  vector<Error> errors;
237  try {
238  elmnt.second->data(m_name.c_str(), ref());
239  //TODO: remove the faulty plugin in case of error?
240  } catch (const Error& e) {
241  errors.emplace_back(e.status(), "for plugin `%s': %s", elmnt.first.c_str(), e.what());
242  } catch (const exception& e) {
243  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s': %s", elmnt.first.c_str(), e.what());
244  } catch (...) {
245  errors.emplace_back(PDI_ERR_SYSTEM, "for plugin `%s'", elmnt.first.c_str());
246  }
247  if ( 1 == errors.size() ) throw Error{errors.front().status(), "While triggering data event `%s' %s", m_name.c_str(), errors.front().what()};
248  if ( !errors.empty() ) {
249  string errmsg = "multiple errors while triggering data event `"+m_name+"':";
250  for ( auto&& err: errors ) {
251  errmsg += " ";
252  errmsg += err.what();
253  }
254  throw Error{PDI_ERR_PLUGIN, "%s", errmsg.c_str()};
255  }
256  }
257 
258  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
259  return result;
260 }

262 void Data_descriptor_impl::release()
263 {
264  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
265  // move reference out of the store
266  if (m_refs.empty() || (m_refs.size()==1 && metadata())) throw Error{PDI_ERR_STATE, "Cannot release a non shared value"};
267 
268  Ref oldref = ref();
269  m_refs.pop();
270 
271  // if only the metadata placeholder ref remains replace it by this one
272  if (metadata() && m_refs.size() == 1) {
273  m_refs.pop();
274  m_refs.emplace(new Ref_holder::Impl<true, false>(oldref));
275  }
276  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
277 }

279 void* Data_descriptor_impl::reclaim()
280 {
281  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
282  if (m_refs.empty() || (m_refs.size()==1 && metadata())) throw Error{PDI_ERR_STATE, "Cannot reclaim a non shared value"};
283 
284  Ref oldref = ref();
285  m_refs.pop();
286 
287  // if only the metadata placeholder ref remains replace it by a copy of this one
288  if (metadata() && m_refs.size() == 1) {
289  m_refs.pop();
290  m_refs.emplace(new Ref_holder::Impl<true, false>(oldref.copy()));
291  }
292 
293  assert((!metadata() || !m_refs.empty()) && "metadata descriptors should always keep a placeholder");
294  // finally release the data behind the ref
295  return oldref.release();
296 }
