50 pybind11::array to_python(Ref r)
51 {
52  ssize_t ndim = 0;
53  vector<ssize_t> starts;
54  vector<ssize_t> shape;
55  vector<ssize_t> strides;
56 
57  const Datatype* subtype = &r.type();
58  while (auto&& array_type = dynamic_cast<const Array_datatype*>(subtype)) {
59  shape.emplace_back(array_type->subsize());
60  if ( ndim ) strides.emplace_back(array_type->size());
61  starts.emplace_back(array_type->start());
62  ++ndim;
63  subtype = &array_type->subtype();
64  }
65  auto&& scalar_type = dynamic_cast<const Scalar_datatype*>(subtype);
66  if ( ndim ) strides.emplace_back(scalar_type->buffersize());
67 
68  pybind11::dtype pytype;
69  switch (scalar_type->kind()) {
70  case Scalar_kind::FLOAT: {
71  switch (scalar_type->datasize()) {
72  case sizeof(float): pytype = pybind11::dtype::of<float>(); break;
73  case sizeof(double): pytype = pybind11::dtype::of<double>(); break;
74  default: throw Error{PDI_ERR_TYPE, "Unable to pass %d bytes floating point value to python", scalar_type->datasize()};
75  }
76  } break;
77  case Scalar_kind::SIGNED: {
78  switch (scalar_type->datasize()) {
79  case sizeof(int8_t): pytype = pybind11::dtype::of<int8_t>(); break;
80  case sizeof(int16_t): pytype = pybind11::dtype::of<int16_t>(); break;
81  case sizeof(int32_t): pytype = pybind11::dtype::of<int32_t>(); break;
82  case sizeof(int64_t): pytype = pybind11::dtype::of<int64_t>(); break;
83  default: throw Error{PDI_ERR_TYPE, "Unable to pass %d bytes integer value to python", scalar_type->datasize()};
84  }
85  } break;
86  case Scalar_kind::UNSIGNED: {
87  switch (scalar_type->datasize()) {
88  case sizeof(uint8_t): pytype = pybind11::dtype::of<uint8_t>(); break;
89  case sizeof(uint16_t): pytype = pybind11::dtype::of<uint16_t>(); break;
90  case sizeof(uint32_t): pytype = pybind11::dtype::of<uint32_t>(); break;
91  case sizeof(uint64_t): pytype = pybind11::dtype::of<uint64_t>(); break;
92  default: throw Error{PDI_ERR_TYPE, "Unable to pass %d bytes unsigned integer value to python", scalar_type->datasize()};
93  }
94  } break;
95  default: throw Error{PDI_ERR_TYPE, "Unable to pass value of unexpected type to python"};
96  }
97 
98  ssize_t cumulated_stride = 1;
99  for (auto&& stride = strides.rbegin(); stride != strides.rend(); ++stride) {
100  *stride *= cumulated_stride;
101  cumulated_stride = *stride;
102  }
103 
104  int64_t* ptr;
105  if (Ref_w r_w{r}) {
106  ptr = static_cast<int64_t*>(r_w.get());
107  } else if (Ref_r r_r{r}) {
108  ptr = static_cast<int64_t*>(const_cast<void*>(r_r.get()));
109  // pybind11::detail::array_descriptor_proxy(result.ptr())->flags &= ~pybind11::detail::npy_api::NPY_ARRAY_WRITEABLE_;
110  }
111  for ( int ii=0; ii<ndim; ++ii ) {
112  ptr += starts[ii]*strides[ii];
113  }
114 
115  Ref* pr = new Ref{r};
116  pybind11::array result = pybind11::array{pytype, move(shape), move(strides), ptr, pybind11::capsule{pr, [](void* pr)
117  {
118  delete static_cast<Ref*>(pr);
119  }}};
120  return result;
121 }

123 Datatype_uptr python_type(pybind11::array& a)
124 {
125  //TODO: handle non C-order arrays
126  vector<size_t> sizes(a.ndim());
127  if (a.ndim()) sizes[0] = a.shape(0);
128  for ( int ii=1; ii<a.ndim(); ++ii ) {
129  sizes[ii] = a.strides(ii-1)/a.strides(ii);
130  }
131  Scalar_kind k;
132  switch (a.dtype().kind()) {
133  case 'c': case 'b': case '?': case 'h': case 'i': case 'l': case 'q': case 'n':
134  k = Scalar_kind::SIGNED;
135  break;
136  case 'B': case 'H': case 'I': case 'L': case 'Q': case 'N':
137  k = Scalar_kind::UNSIGNED;
138  break;
139  case 'e': case 'f': case 'd': case 'g':
140  k = Scalar_kind::FLOAT;
141  break;
142  case 'P':
143  k = Scalar_kind::ADDRESS;
144  break;
145  default:
146  throw Error{PDI_ERR_IMPL, "Unexpected python type descriptor: %c", a.dtype().kind()};
147  }
148 
149  Datatype_uptr result{new Scalar_datatype{k, static_cast<size_t>(a.dtype().itemsize())}};
150  for ( int ii=a.ndim()-1; ii>=0; --ii ) {
151  result.reset(new Array_datatype{move(result), sizes[ii], 0, static_cast<size_t>(a.shape(ii))});
152  }
153  return result;
154 }
